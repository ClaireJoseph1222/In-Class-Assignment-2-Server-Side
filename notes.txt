camelCase routes naming conventions


- POST /api/auth/register
    - body: { "username": string, "email": string, "password": string }
- POST /api/auth/login
    - body: { "email": string, "password": string }
    - response: { "token": string, "user": { id, username, email } }
    - client must send Authorization: Bearer <token> for protected routes
- POST /api/games/complete
    - headers: Authorization: Bearer <token>
    - body: { "rounds": number, "userWin": boolean }
    - server sets date_completed to NOW()
- GET /api/games/history
    - headers: Authorization: Bearer <token>
    - response: DbGame[] (id, date_completed, user_win, rounds, user_id)



CREATE TABLE games (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    rounds INT NOT NULL,
    did_win BOOLEAN NOT NULL,
    finished_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);


CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL
);
